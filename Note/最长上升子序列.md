
## 动态规划 $O(n^2)$

```
dp[i] 前i个元素的最长上升子序列
转移方程 dp[i]=max(dp[j]+1,dp[i]);
``` 

```c++
int maxx = 0;
    for (int i = 0; i < n; i++)
    {
        dp[i] = 1;
        for (int j = 0; j < i; j++)
        {
            if (arr[i] > arr[j])
            {
                dp[i] = max(dp[j] + 1, dp[i]);
            }
        }
        maxx = max(maxx, dp[i]);
    }
```

```c++
int idx = 0;
    for (int i = 0; i < n; i++)
    {
        int l = 0, r = idx;
        while (l < r)
        {
	        // 寻找arr[i]的位置（或插入的位置）
            int mid = l + ((r - l) >> 1);
            if (dp[mid] == arr[i])
            {
                l = r = mid;
            }
            else if (dp[mid] < arr[i])
            {
                l = mid + 1;
            }
            else if (dp[mid] > arr[i])
            {
                r = mid;
            }
        }
        dp[l] = arr[i];
        if (l == idx)
        {
            idx++;
        }
    }
```
